    $(document).ready(function() { getNews("all"); });

    function getNews(category) {
        var fd = new FormData();
        if(category != 'all'){
            fd.append("category", category);
        }
        $.ajax({
            url : "/api/getNewsList.do",
            type : "post",
            data: fd,
            contentType: false,
            processData: false,
            success : function(res) {
                if (res.resultCode != "200") {
                    alert('뉴스 로딩에 실패했습니다. 페이지를 새로 고침 해주세요');
                    return;
                }
                updateNews(res.data);
            },
            error: function(response){
                console.log(response);
            }
        });
    }

    function updateNews(newsData) {
        $('#news-'+newsData.category).empty();
        $('#news-'+newsData.category).append(getNewsBody(newsData.list));
    }

    function getNewsBody(newsData) {
        var html = '';

        var main_news = newsData[0];
        if (!main_news) {
          return;
        }

        html += '<li class="l-box">';
        html += '<a href="/news_view.do?orderBy=seq&viewOption=album&seq=' + main_news['seq'] + '" class="f-wrap">';
        html += '<div class="img-box left">';

        if(main_news['thumb'] == null){
            html += '<img src="/assets/images/no_thumbnail.png" width="100%">';
        }
        else{
            if(main_news['thumb'].indexOf("press.kleague.com") > 0) {
                //console.log(item.thumb)
                html += '<img src="' + main_news['thumb'] + '" alt="">';
            } else {
                html += '<img src="https://d2tfp74nsbbrkr.cloudfront.net' + main_news['thumb'] + '" alt="">';
            }
        }
        html += '</div>';
        html += '<div class="txt-box left">';
        html += '<span class="sort">' + main_news['category'].toUpperCase() + '</span>';
        html += '<strong class="name">' + main_news['title'] + '</strong>';
        html += '<span class="s-txt">' + main_news['title'] + '</span>';
        html += '<span class="btn-more icon"><span class="hidden">더보기</span></span>';
        html += '</div>';
//         html += '</a>';
        html += '</li>';

        for(var i = 1; i < 4; i++) {
            news = newsData[i];

            if (i == 3) {
                html += '<li class="last">';
            }else {
                html += '<li>';
            }

            html += '<a href="/news_view.do?orderBy=seq&viewOption=album&seq=' + news['seq'] + '">';
            html += '<div class="img-box">';

            if(news['thumb']==null){
                html += '<img src="/assets/images/no_thumbnail.png" width="100%">';
            }
            else {
                if(news['thumb'].indexOf("press.kleague.com") > 0) {
                    html += '<img src="' + news['thumb'] + '" alt="">';
                } else {
                    html += '<img src="https://d2tfp74nsbbrkr.cloudfront.net' + news['thumb'] + '" alt="">';
                }
            }
            html += '<span class="sort">' + news['category'].toUpperCase() + '</span>';
            html += '<strong class="name">' + news['title'] + '</strong>';
            html += '</div>';
            html += '</a>';
            html += '</li>';
        }
        return html;
    }

    document.addEventListener("DOMContentLoaded", function () {

        if (!todayStop || currentDate.getTime() > parseInt(todayStop)) {
            popup.style.display = "block";

function closePopup() {
    const popup = document.getElementById("noticePop");
    const checkbox = document.getElementById("chk");

    if (checkbox.checked) {
        const now = new Date();
        const expireTime = now.getTime() + 24 * 60 * 60 * 1000; // 1일 후 만료 시간
        localStorage.setItem("todayStop", expireTime);
    }

    popup.style.display = "none";
}
